### Database Schema Addition

```sql
CREATE TABLE governance_proposals (
  id SERIAL PRIMARY KEY,
  nation_id INTEGER REFERENCES rainbow_roads(id),
  title VARCHAR(200) NOT NULL,
  description TEXT NOT NULL,
  domain VARCHAR(20) NOT NULL, -- ether, fire, earth, etc.
  status VARCHAR(20) DEFAULT 'active',
  created_by TEXT NOT NULL, -- wallet
  created_at TIMESTAMP DEFAULT NOW(),
  voting_ends_at TIMESTAMP NOT NULL,
  execution_data JSONB
);

CREATE TABLE governance_votes (
  id SERIAL PRIMARY KEY,
  proposal_id INTEGER REFERENCES governance_proposals(id),
  voter_wallet TEXT NOT NULL,
  vote VARCHAR(10) NOT NULL, -- yes, no, abstain
  voting_power INTEGER NOT NULL, -- degrees in relevant element
  voted_at TIMESTAMP DEFAULT NOW(),
  UNIQUE(proposal_id, voter_wallet)
);
```
